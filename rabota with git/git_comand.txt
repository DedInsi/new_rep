cd /Desktop/ - переход к папки

git --version - просмотр версии git

ls - показывает папки 

git init - создает папку .git в файле для отслеживания

git status - проверяет статус папок отслеживаемые или нет и есть ли комит или нет 

git add (. - все файлы добавляет / можно указать просто название папки которое нужно отслеживать) - добавляет статус папки в отслеживания 

git commit -m "(тут пишеться комментарий без скобочек в кавычках)" - добавляет комментарий что изменили 

git log - выводит все комментарии которые писались

git branch (название новой ветки) - добавляет новую ветку

git branch -a - показывает какие ветки есть 

git checkout (название ветки)- переходит на другую ветку 

git checkout -b (название ветки) - создает и сразу переходит к этой ветки

git branch -d (Название ветки) - удаляет ветку 

git merge (название ветки с которой нужно провести слияние) - слияние веток, должны находиться 
в ветке В КОТОРУЮ нужно сделать слияние например если в master то и в ней должны находиться 

git remote add origin (ссылка на директорю на github) - подключение удаленного репозитория

git push -u origin master - копирует в git hib файлы в репозиторий master
